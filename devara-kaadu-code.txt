import { useState } from "react";
import { useLocation } from "wouter";
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { z } from "zod";
import { ArrowLeft, Camera, X, ImageIcon, Loader2 } from "lucide-react";
import { Link } from "wouter";
import { useCreateAlert, useListGroves, getListAlertsQueryKey } from "@workspace/api-client-react";
import { useUpload } from "@workspace/object-storage-web";
import { useQueryClient } from "@tanstack/react-query";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Form, FormControl, FormField, FormItem, FormLabel, FormMessage } from "@/components/ui/form";
import { Input } from "@/components/ui/input";
import { Textarea } from "@/components/ui/textarea";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { useToast } from "@/hooks/use-toast";

const formSchema = z.object({
  groveId: z.string().optional(),
  alertType: z.string().min(1, "Alert type is required"),
  description: z.string().min(10, "Please provide a detailed description (at least 10 characters)"),
  reportedBy: z.string().optional(),
  location: z.string().optional(),
});

type FormValues = z.infer<typeof formSchema>;

const ALERT_TYPES = [
  { value: "tree_cutting", label: "Tree Cutting" },
  { value: "waste_dumping", label: "Waste Dumping" },
  { value: "encroachment", label: "Encroachment" },
  { value: "fire", label: "Fire" },
  { value: "other", label: "Other" },
];

export default function AlertNew() {
  const [, setLocation] = useLocation();
  const { toast } = useToast();
  const queryClient = useQueryClient();
  const [uploadedImagePath, setUploadedImagePath] = useState<string | null>(null);
  const [imagePreviewUrl, setImagePreviewUrl] = useState<string | null>(null);

  const { data: groves } = useListGroves({});

  const { uploadFile, isUploading, progress } = useUpload({
    onSuccess: (response) => {
      setUploadedImagePath(response.objectPath);
      toast({ title: "Photo uploaded", description: "Your photo has been attached to this alert." });
    },
    onError: (err) => {
      toast({ title: "Upload failed", description: err.message, variant: "destructive" });
    },
  });

  const createAlert = useCreateAlert({
    mutation: {
      onSuccess: () => {
        queryClient.invalidateQueries({ queryKey: getListAlertsQueryKey() });
        toast({ title: "Alert reported", description: "Thank you for helping protect our sacred groves." });
        setLocation("/alerts");
      },
      onError: () => {
        toast({ title: "Failed to submit alert", variant: "destructive" });
      },
    },
  });

  const form = useForm<FormValues>({
    resolver: zodResolver(formSchema),
    defaultValues: {
      groveId: "",
      alertType: "",
      description: "",
      reportedBy: "",
      location: "",
    },
  });

  const handlePhotoSelect = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;
    const localUrl = URL.createObjectURL(file);
    setImagePreviewUrl(localUrl);
    await uploadFile(file);
  };

  const handleRemovePhoto = () => {
    setUploadedImagePath(null);
    setImagePreviewUrl(null);
  };

  const onSubmit = (values: FormValues) => {
    const selectedGrove = groves?.find((g) => String(g.id) === values.groveId);
    createAlert.mutate({
      data: {
        groveId: values.groveId ? parseInt(values.groveId, 10) : null,
        groveName: selectedGrove?.name ?? null,
        alertType: values.alertType,
        description: values.description,
        reportedBy: values.reportedBy || null,
        location: values.location || null,
        imageUrl: uploadedImagePath ? `/api/storage${uploadedImagePath}` : null,
      },
    });
  };

  return (
    <div className="px-4 py-4 max-w-2xl mx-auto">
      <div className="flex items-center gap-3 mb-5">
        <Link href="/alerts" className="flex items-center justify-center w-8 h-8 rounded-full bg-muted hover:bg-muted/70 transition-colors">
          <ArrowLeft className="h-4 w-4 text-foreground" />
        </Link>
        <div>
          <h1 className="text-lg font-serif font-bold text-foreground leading-tight">Report Alert</h1>
          <p className="text-[11px] text-muted-foreground">Flag a conservation threat</p>
        </div>
      </div>

      <Card>
        <CardHeader>
          <CardTitle className="font-serif text-xl">Alert Details</CardTitle>
        </CardHeader>
        <CardContent>
          <Form {...form}>
            <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
              <FormField
                control={form.control}
                name="alertType"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Type of Threat</FormLabel>
                    <Select onValueChange={field.onChange} value={field.value}>
                      <FormControl>
                        <SelectTrigger data-testid="select-alert-type">
                          <SelectValue placeholder="Select threat type" />
                        </SelectTrigger>
                      </FormControl>
                      <SelectContent>
                        {ALERT_TYPES.map((t) => (
                          <SelectItem key={t.value} value={t.value}>{t.label}</SelectItem>
                        ))}
                      </SelectContent>
                    </Select>
                    <FormMessage />
                  </FormItem>
                )}
              />

              <FormField
                control={form.control}
                name="groveId"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Sacred Grove (optional)</FormLabel>
                    <Select onValueChange={field.onChange} value={field.value}>
                      <FormControl>
                        <SelectTrigger data-testid="select-grove">
                          <SelectValue placeholder="Select a grove if known" />
                        </SelectTrigger>
                      </FormControl>
                      <SelectContent>
                        <SelectItem value="unknown">Unknown / Not listed</SelectItem>
                        {groves?.map((g) => (
                          <SelectItem key={g.id} value={String(g.id)}>{g.name} — {g.district}</SelectItem>
                        ))}
                      </SelectContent>
                    </Select>
                    <FormMessage />
                  </FormItem>
                )}
              />

              <FormField
                control={form.control}
                name="description"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Description</FormLabel>
                    <FormControl>
                      <Textarea
                        data-testid="input-alert-description"
                        placeholder="Describe what you observed. Include as much detail as possible — what happened, when, and the extent of damage."
                        className="min-h-[120px]"
                        {...field}
                      />
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}
              />

              {/* Photo Upload */}
              <div className="space-y-2">
                <label className="text-sm font-medium leading-none">
                  Attach Photo (optional)
                </label>
                {imagePreviewUrl ? (
                  <div className="relative rounded-lg overflow-hidden border bg-muted">
                    <img
                      src={imagePreviewUrl}
                      alt="Alert photo preview"
                      className="w-full max-h-56 object-cover"
                    />
                    {isUploading && (
                      <div className="absolute inset-0 bg-background/70 flex flex-col items-center justify-center gap-2">
                        <Loader2 className="h-6 w-6 animate-spin text-primary" />
                        <span className="text-sm font-medium">{progress}%</span>
                      </div>
                    )}
                    {!isUploading && (
                      <button
                        type="button"
                        onClick={handleRemovePhoto}
                        className="absolute top-2 right-2 bg-background/80 backdrop-blur rounded-full p-1 hover:bg-background transition-colors"
                        aria-label="Remove photo"
                      >
                        <X className="h-4 w-4" />
                      </button>
                    )}
                    {uploadedImagePath && !isUploading && (
                      <div className="absolute bottom-2 left-2 bg-green-600/90 text-white text-xs px-2 py-1 rounded-full">
                        Uploaded
                      </div>
                    )}
                  </div>
                ) : (
                  <label
                    htmlFor="photo-upload"
                    className="flex flex-col items-center justify-center gap-3 border-2 border-dashed border-border rounded-lg p-8 cursor-pointer hover:border-primary/50 hover:bg-primary/5 transition-colors"
                  >
                    <div className="flex items-center justify-center w-12 h-12 rounded-full bg-primary/10">
                      <Camera className="h-6 w-6 text-primary" />
                    </div>
                    <div className="text-center">
                      <p className="text-sm font-medium text-foreground">Click to add a photo</p>
                      <p className="text-xs text-muted-foreground mt-1">JPG, PNG or WEBP up to 10MB</p>
                    </div>
                    <input
                      id="photo-upload"
                      type="file"
                      accept="image/jpeg,image/png,image/webp"
                      className="sr-only"
                      onChange={handlePhotoSelect}
                      data-testid="input-alert-photo"
                    />
                  </label>
                )}
              </div>

              <FormField
                control={form.control}
                name="location"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Location (optional)</FormLabel>
                    <FormControl>
                      <Input data-testid="input-alert-location" placeholder="e.g. Northern edge of grove, near the stream" {...field} />
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}
              />

              <FormField
                control={form.control}
                name="reportedBy"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Your Name (optional)</FormLabel>
                    <FormControl>
                      <Input data-testid="input-alert-reporter" placeholder="Your name" {...field} />
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}
              />

              <div className="flex gap-3 pt-2">
                <Button
                  type="submit"
                  data-testid="button-submit-alert"
                  disabled={createAlert.isPending || isUploading}
                  className="flex-1"
                >
                  {createAlert.isPending ? "Submitting..." : isUploading ? "Uploading photo..." : "Submit Alert"}
                </Button>
                <Button type="button" variant="outline" asChild>
                  <Link href="/alerts">Cancel</Link>
                </Button>
              </div>
            </form>
          </Form>
        </CardContent>
      </Card>
    </div>
  );
}

import { Link } from "wouter";
import { useState } from "react";
import { AlertTriangle, Plus, ShieldCheck, ChevronRight } from "lucide-react";
import { useListAlerts, useUpdateAlert, getListAlertsQueryKey } from "@workspace/api-client-react";
import { useQueryClient } from "@tanstack/react-query";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { useToast } from "@/hooks/use-toast";

const statusColor: Record<string, string> = {
  reported: "bg-red-100 text-red-800",
  investigating: "bg-amber-100 text-amber-800",
  resolved: "bg-emerald-100 text-emerald-800",
};

const alertTypeLabels: Record<string, string> = {
  tree_cutting: "Tree Cutting",
  waste_dumping: "Waste Dumping",
  encroachment: "Encroachment",
  fire: "Fire",
  other: "Other",
};

const STATUS_FILTERS = [
  { value: "", label: "All" },
  { value: "reported", label: "Reported" },
  { value: "investigating", label: "Investigating" },
  { value: "resolved", label: "Resolved" },
];

export default function Alerts() {
  const [statusFilter, setStatusFilter] = useState("");
  const { toast } = useToast();
  const queryClient = useQueryClient();

  const { data: alerts, isLoading } = useListAlerts({
    status: statusFilter as "reported" | "investigating" | "resolved" | undefined || undefined,
  });

  const updateAlert = useUpdateAlert({
    mutation: {
      onSuccess: () => {
        queryClient.invalidateQueries({ queryKey: getListAlertsQueryKey() });
        toast({ title: "Status updated" });
      },
    },
  });

  const handleStatusChange = (id: number, status: "reported" | "investigating" | "resolved") => {
    updateAlert.mutate({ id, data: { status } });
  };

  return (
    <div className="flex flex-col">
      {/* Header */}
      <div className="px-4 pt-5 pb-3 flex items-start justify-between">
        <div>
          <h1 className="text-xl font-serif font-bold text-foreground">Conservation Alerts</h1>
          <p className="text-xs text-muted-foreground mt-0.5">Track and respond to threats</p>
        </div>
        <Link
          href="/alerts/new"
          data-testid="button-report-alert"
          className="flex items-center gap-1.5 bg-primary text-primary-foreground text-xs font-semibold px-3 py-2 rounded-xl hover:bg-primary/90 transition-colors active:scale-95"
        >
          <Plus className="h-3.5 w-3.5" />
          Report
        </Link>
      </div>

      {/* Status filter pills */}
      <div className="px-4 pb-3 flex gap-2 overflow-x-auto scrollbar-none">
        {STATUS_FILTERS.map((f) => (
          <button
            key={f.value}
            onClick={() => setStatusFilter(f.value)}
            data-testid={f.value ? `filter-${f.value}` : "filter-all"}
            className={`shrink-0 px-3 py-1 rounded-full text-xs font-medium transition-colors ${
              statusFilter === f.value ? "bg-primary text-primary-foreground" : "bg-muted text-muted-foreground"
            }`}
          >
            {f.label}
          </button>
        ))}
      </div>

      {/* List */}
      <div className="px-4 space-y-3">
        {isLoading ? (
          Array.from({ length: 4 }).map((_, i) => (
            <div key={i} className="h-20 rounded-2xl bg-muted animate-pulse" />
          ))
        ) : alerts?.length === 0 ? (
          <div className="flex flex-col items-center py-20 text-muted-foreground gap-3">
            <ShieldCheck className="h-12 w-12 text-emerald-400" />
            <p className="text-sm font-medium text-emerald-700">The groves are safe.</p>
          </div>
        ) : (
          alerts?.map((alert) => (
            <div
              key={alert.id}
              data-testid={`card-alert-${alert.id}`}
              className="bg-card border border-border/60 rounded-2xl overflow-hidden"
            >
              <div className="flex items-center gap-3 px-4 pt-3.5 pb-2">
                <div className="w-8 h-8 rounded-full bg-destructive/10 flex items-center justify-center shrink-0">
                  <AlertTriangle className="h-4 w-4 text-destructive" />
                </div>
                <div className="flex-1 min-w-0">
                  <p className="text-sm font-semibold text-foreground">
                    {alertTypeLabels[alert.alertType] ?? alert.alertType}
                  </p>
                  {alert.groveName && (
                    <p className="text-[10px] text-muted-foreground">{alert.groveName}</p>
                  )}
                </div>
                <span className={`text-[10px] font-semibold px-2 py-0.5 rounded-full capitalize shrink-0 ${statusColor[alert.status] ?? "bg-gray-100 text-gray-700"}`}>
                  {alert.status}
                </span>
              </div>
              <p className="px-4 pb-2 text-[11px] text-foreground/70 line-clamp-2 leading-relaxed">{alert.description}</p>
              <div className="px-4 pb-3 flex items-center justify-between gap-2 border-t border-border/50 pt-2.5">
                <p className="text-[10px] text-muted-foreground">
                  {new Date(alert.createdAt).toLocaleDateString()}
                  {alert.location && ` · ${alert.location}`}
                </p>
                <Select
                  value={alert.status}
                  onValueChange={(v) => handleStatusChange(alert.id, v as "reported" | "investigating" | "resolved")}
                >
                  <SelectTrigger data-testid={`select-alert-status-${alert.id}`} className="w-36 h-7 text-[11px]">
                    <SelectValue />
                  </SelectTrigger>
                  <SelectContent>
                    <SelectItem value="reported">Reported</SelectItem>
                    <SelectItem value="investigating">Investigating</SelectItem>
                    <SelectItem value="resolved">Resolved</SelectItem>
                  </SelectContent>
                </Select>
              </div>
            </div>
          ))
        )}
      </div>
    </div>
  );
}

import { useRoute, Link } from "wouter";
import { ArrowLeft, MapPin, TreePine, Leaf, BookOpen, AlertTriangle, ShieldCheck } from "lucide-react";
import { useGetGrove, useListGroveSpecies, useListGroveLegends, useListGroveAlerts, getGetGroveQueryKey, getListGroveSpeciesQueryKey, getListGroveLegendsQueryKey, getListGroveAlertsQueryKey } from "@workspace/api-client-react";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";

const statusColor: Record<string, string> = {
  "Well-preserved": "bg-green-100 text-green-800",
  "Threatened": "bg-amber-100 text-amber-800",
  "Critical": "bg-red-100 text-red-800",
};

const speciesStatusColor: Record<string, string> = {
  "Endangered": "bg-red-100 text-red-800",
  "Vulnerable": "bg-amber-100 text-amber-800",
  "Least Concern": "bg-green-100 text-green-800",
};

const alertStatusColor: Record<string, string> = {
  "reported": "bg-red-100 text-red-800",
  "investigating": "bg-amber-100 text-amber-800",
  "resolved": "bg-green-100 text-green-800",
};

export default function GroveDetail() {
  const [, params] = useRoute("/groves/:id");
  const id = parseInt(params?.id ?? "0", 10);

  const { data: grove, isLoading } = useGetGrove(id, { query: { enabled: !!id, queryKey: getGetGroveQueryKey(id) } });
  const { data: species } = useListGroveSpecies(id, { query: { enabled: !!id, queryKey: getListGroveSpeciesQueryKey(id) } });
  const { data: legends } = useListGroveLegends(id, { query: { enabled: !!id, queryKey: getListGroveLegendsQueryKey(id) } });
  const { data: alerts } = useListGroveAlerts(id, { query: { enabled: !!id, queryKey: getListGroveAlertsQueryKey(id) } });

  if (isLoading) {
    return (
      <div className="container mx-auto px-4 py-10">
        <div className="animate-pulse space-y-4">
          <div className="h-8 w-48 bg-muted rounded" />
          <div className="h-64 bg-muted rounded-xl" />
          <div className="h-6 w-2/3 bg-muted rounded" />
          <div className="h-4 w-full bg-muted rounded" />
        </div>
      </div>
    );
  }

  if (!grove) {
    return (
      <div className="container mx-auto px-4 py-20 text-center">
        <TreePine className="h-12 w-12 mx-auto mb-4 opacity-30" />
        <p className="text-muted-foreground text-lg">Grove not found.</p>
        <Button variant="link" asChild className="mt-4"><Link href="/groves">Back to directory</Link></Button>
      </div>
    );
  }

  const traditionalLegends = legends?.filter((l) => l.legendType === "traditional") ?? [];
  const scientificLegends = legends?.filter((l) => l.legendType === "scientific") ?? [];

  return (
    <div className="container mx-auto px-4 py-10 max-w-4xl">
      <Button variant="ghost" asChild className="mb-6 -ml-2">
        <Link href="/groves" data-testid="link-back-to-groves">
          <ArrowLeft className="h-4 w-4 mr-2" />Back to Groves
        </Link>
      </Button>

      <div className="mb-8">
        {grove.imageUrl && (
          <div className="w-full h-64 rounded-xl overflow-hidden mb-6">
            <img src={grove.imageUrl} alt={grove.name} className="object-cover w-full h-full" />
          </div>
        )}
        <div className="flex flex-wrap items-start justify-between gap-4 mb-4">
          <div>
            <h1 className="text-4xl font-serif font-bold mb-2" data-testid="text-grove-name">{grove.name}</h1>
            <div className="flex flex-wrap items-center gap-3 text-muted-foreground">
              <span className="flex items-center gap-1.5"><MapPin className="h-4 w-4" />{grove.district}, {grove.state}</span>
              {grove.area && <span className="flex items-center gap-1.5"><Leaf className="h-4 w-4" />{grove.area}</span>}
            </div>
          </div>
          <div className="flex flex-wrap gap-2">
            <Badge className="text-sm" variant="secondary">{grove.groveType}</Badge>
            <span className={`px-3 py-1 text-sm font-medium rounded-full ${statusColor[grove.conservationStatus] ?? "bg-gray-100 text-gray-800"}`}>
              {grove.conservationStatus}
            </span>
          </div>
        </div>
        {grove.deity && (
          <p className="text-sm text-muted-foreground mb-4 italic">Presiding deity: {grove.deity}</p>
        )}
        <p className="text-foreground/80 leading-relaxed text-base">{grove.description}</p>
      </div>

      <Tabs defaultValue="legends" data-testid="tabs-grove-detail">
        <TabsList className="mb-6">
          <TabsTrigger value="legends" data-testid="tab-legends">
            <BookOpen className="h-4 w-4 mr-2" />Myths & Legends
          </TabsTrigger>
          <TabsTrigger value="species" data-testid="tab-species">
            <Leaf className="h-4 w-4 mr-2" />Species ({species?.length ?? 0})
          </TabsTrigger>
          <TabsTrigger value="alerts" data-testid="tab-alerts">
            <AlertTriangle className="h-4 w-4 mr-2" />Alerts ({alerts?.length ?? 0})
          </TabsTrigger>
        </TabsList>

        <TabsContent value="legends">
          {legends?.length === 0 ? (
            <p className="text-muted-foreground text-center py-10">No legends documented yet.</p>
          ) : (
            <div className="space-y-8">
              {traditionalLegends.length > 0 && (
                <div>
                  <h3 className="text-lg font-serif font-semibold mb-4 flex items-center gap-2">
                    <span className="w-2 h-2 rounded-full bg-amber-500 inline-block" />Traditional Beliefs
                  </h3>
                  <div className="space-y-4">
                    {traditionalLegends.map((legend) => (
                      <Card key={legend.id} data-testid={`card-legend-${legend.id}`} className="border-amber-200/60 bg-amber-50/30 dark:bg-amber-950/10">
                        <CardHeader className="pb-2">
                          <CardTitle className="font-serif text-lg">{legend.title}</CardTitle>
                          {legend.deity && <p className="text-xs text-muted-foreground">Deity: {legend.deity}</p>}
                        </CardHeader>
                        <CardContent>
                          <p className="text-sm leading-relaxed text-foreground/80 italic">"{legend.content}"</p>
                          {legend.source && <p className="text-xs text-muted-foreground mt-3">Source: {legend.source}</p>}
                        </CardContent>
                      </Card>
                    ))}
                  </div>
                </div>
              )}
              {scientificLegends.length > 0 && (
                <div>
                  <h3 className="text-lg font-serif font-semibold mb-4 flex items-center gap-2">
                    <ShieldCheck className="h-4 w-4 text-primary" />Scientific Facts
                  </h3>
                  <div className="space-y-4">
                    {scientificLegends.map((legend) => (
                      <Card key={legend.id} data-testid={`card-science-${legend.id}`} className="border-primary/20 bg-primary/5">
                        <CardHeader className="pb-2">
                          <CardTitle className="font-serif text-lg">{legend.title}</CardTitle>
                        </CardHeader>
                        <CardContent>
                          <p className="text-sm leading-relaxed text-foreground/80">{legend.content}</p>
                          {legend.source && <p className="text-xs text-muted-foreground mt-3">Source: {legend.source}</p>}
                        </CardContent>
                      </Card>
                    ))}
                  </div>
                </div>
              )}
            </div>
          )}
        </TabsContent>

        <TabsContent value="species">
          {species?.length === 0 ? (
            <p className="text-muted-foreground text-center py-10">No species documented yet.</p>
          ) : (
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              {species?.map((s) => (
                <Card key={s.id} data-testid={`card-species-${s.id}`}>
                  <CardHeader className="pb-2">
                    <div className="flex justify-between items-start gap-2">
                      <div>
                        <CardTitle className="font-serif text-base">{s.commonName}</CardTitle>
                        <p className="text-xs italic text-muted-foreground">{s.scientificName}</p>
                      </div>

import { createContext, useContext, useState, useEffect, useCallback } from "react";

export interface AuthUser {
  id: number;
  username: string;
  displayName: string | null;
  role: string;
}

interface AuthContextValue {
  user: AuthUser | null;
  isLoading: boolean;
  login: (username: string, password: string) => Promise<{ ok: boolean; error?: string }>;
  logout: () => Promise<void>;
}

const AuthContext = createContext<AuthContextValue | null>(null);

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const [user, setUser] = useState<AuthUser | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  const fetchMe = useCallback(async () => {
    try {
      const res = await fetch("/api/auth/me", { credentials: "include" });
      if (res.ok) {
        const data = await res.json();
        setUser(data.user);
      } else {
        setUser(null);
      }
    } catch {
      setUser(null);
    } finally {
      setIsLoading(false);
    }
  }, []);

  useEffect(() => {
    fetchMe();
  }, [fetchMe]);

  const login = useCallback(async (username: string, password: string) => {
    const res = await fetch("/api/auth/login", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      credentials: "include",
      body: JSON.stringify({ username, password }),
    });
    if (res.ok) {
      const data = await res.json();
      setUser(data.user);
      return { ok: true };
    }
    const data = await res.json().catch(() => ({ error: "Login failed" }));
    return { ok: false, error: data.error ?? "Login failed" };
  }, []);

  const logout = useCallback(async () => {
    await fetch("/api/auth/logout", { method: "POST", credentials: "include" });
    setUser(null);
  }, []);

  return (
    <AuthContext.Provider value={{ user, isLoading, login, logout }}>
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth() {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error("useAuth must be used inside AuthProvider");
  return ctx;
}

const router = Router();

router.get("/auth/me", (req, res) => {
  if (!req.session.userId) {
    res.status(401).json({ error: "Not authenticated" });
    return;
  }
  res.json({
    user: {
      id: req.session.userId,
      username: req.session.username,
      displayName: req.session.displayName ?? null,
      role: req.session.role,
    },
  });
});

router.post("/auth/login", async (req, res) => {
  const { username, password } = req.body as { username?: string; password?: string };

  if (!username || !password) {
    res.status(400).json({ error: "Username and password are required" });
    return;
  }

  const [user] = await db
    .select()
    .from(usersTable)
    .where(eq(usersTable.username, username.trim().toLowerCase()))
    .limit(1);

  if (!user) {
    res.status(401).json({ error: "Invalid username or password" });
    return;
  }

  const passwordMatch = await bcrypt.compare(password, user.passwordHash);
  if (!passwordMatch) {
    res.status(401).json({ error: "Invalid username or password" });
    return;
  }

  req.session.userId = user.id;
  req.session.username = user.username;
  req.session.displayName = user.displayName ?? undefined;
  req.session.role = user.role;

  res.json({
    user: {
      id: user.id,
      username: user.username,
      displayName: user.displayName,
      role: user.role,
    },
  });
});

router.post("/auth/logout", (req, res) => {
  req.session.destroy(() => {
    res.clearCookie("sid");
    res.json({ ok: true });
  });
});

 * Request a presigned URL for file upload.
 * The client sends JSON metadata (name, size, contentType) — NOT the file.

    return;
  }

  try {

    const uploadURL = await objectStorageService.getObjectEntityUploadURL();
    const objectPath = objectStorageService.normalizeObjectEntityPath(uploadURL);

    res.json(
      RequestUploadUrlResponse.parse({
        uploadURL,
        objectPath,
        metadata: { name, size, contentType },
      }),
    );
  } catch (error) {
    req.log.error({ err: error }, "Error generating upload URL");
    res.status(500).json({ error: "Failed to generate upload URL" });

/**
 * GET /storage/public-objects/*
 *
 * Serve public assets from PUBLIC_OBJECT_SEARCH_PATHS.
 * These are unconditionally public — no authentication or ACL checks.
 * IMPORTANT: Always provide this endpoint when object storage is set up.
 */
router.get("/storage/public-objects/*filePath", async (req: Request, res: Response) => {
  try {
    const raw = req.params.filePath;
    const filePath = Array.isArray(raw) ? raw.join("/") : raw;
    const file = await objectStorageService.searchPublicObject(filePath);
    if (!file) {
      res.status(404).json({ error: "File not found" });
      return;

    const response = await objectStorageService.downloadObject(file);

    if (response.body) {
      const nodeStream = Readable.fromWeb(response.body as ReadableStream<Uint8Array>);
      nodeStream.pipe(res);
    } else {
      res.end();
    }
  } catch (error) {
    req.log.error({ err: error }, "Error serving public object");
    res.status(500).json({ error: "Failed to serve public object" });
  }
});
